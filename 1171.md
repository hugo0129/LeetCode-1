class Solution {
    public ListNode removeZeroSumSublists(ListNode head) {
        ListNode dummy = new ListNode(-1);
        dummy.next = head;
        ListNode first = dummy, second;
        while (first != null) {
            second = first.next;
            int sum = 0;
            while (second != null) {
                sum += second.val;
                second = second.next;
                if(sum == 0) {
                    first.next = second;
                    sum = 0;
                }
            }
            first = first.next;
        }
        return dummy.next;
    }
}
