class Solution {
    public int findKthNumber(int m, int n, int k) {
        if (k <= 0 || m <= 0 || n <= 0 || k > m * n) throw new IllegalArgumentException("invalid param");

        int left = 1, right = m * n;
        while (left < right) {
            int mid = left + (right - left) / 2;
            // right为第k个元素所在层
            if (count(mid, m, n) >= k) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return left;
    }

    private int count(int target, int m, int n) {
        int count = 0;
        if (m > n) {
            int temp = m;
            m = n;
            n = temp;
        }
        int row = 1, col = n;
        while (row <= m && col >= 0) {
            if (row * col > target) {
                col--;
            } else {
                count += col;
                row++;
            }
        }
        return count;
    }
}
