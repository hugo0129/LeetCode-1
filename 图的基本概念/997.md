class Solution {
    public int findJudge(int N, int[][] trust) {
        if (N <= 0 || trust == null) return -1;
        // 统计出度
        int[] outDegree = new int[N];
        int[] inDegree = new int[N];
        for (int[] edge : trust) {
            outDegree[edge[0] - 1]++;
            inDegree[edge[1] - 1]++;
        }
        int idx = -1;
        for (int i = 0; i < N; i++) {
            if (outDegree[i] == 0 && inDegree[i] == (N - 1)) {
                if (idx == -1) idx = i + 1;
                else return -1;
            }
        }
        return idx;
    }
}
