class Solution {
    public int search(ArrayReader reader, int target) {
        if(target >= 10_000 || target <= -10_000) return -1;
        int left = 0, right = 19_999;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if(reader.get(mid) == target) return mid;
            if(reader.get(mid) > target) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        return -1;
    }
}
