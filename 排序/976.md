class Solution {
    public int largestPerimeter(int[] A) {
        if(A == null || A.length < 3) {
            return 0;
        }
        Arrays.sort(A);
        int a = A.length - 3, b = A.length - 2, c = A.length - 1;
        while (a >= 0) {
            // 满足条件
            if(A[a] + A[b] > A[c]) {
               return A[a] + A[b] + A[c];
            }
            else {
                a--;
                b--;
                c--;
            }
        }
        return 0;
    }
}
