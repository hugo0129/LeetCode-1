class Solution {
    private static Map<Character, Integer> roman = new HashMap<>();
    static {
        roman.put('I', 1);
        roman.put('V', 5);
        roman.put('X', 10);
        roman.put('L', 50);
        roman.put('C', 100);
        roman.put('D', 500);
        roman.put('M', 1000);
    }

    public int romanToInt(String s) {
        int num = 0;
        char pre = '#';
        for(char c : s.toCharArray()) {
            if(pre != '#' && roman.get(pre) < roman.get(c)) {
                num += roman.get(c) - 2 * roman.get(pre);
            } else {
                num += roman.get(c);
            }
            pre = c;
        }
        return num;
    }
}
