class Solution {
    public int missingElement(int[] nums, int k) {
        if(k <= 0 || nums == null || nums.length == 0) throw new IllegalArgumentException("invalid param");

        int min = nums[0];
        int left = 0, right = nums.length;
        while (left < right) {
            int mid = left + (right - left) / 2;
            // mid前面的区间缺失超过k个值
            if(nums[mid] - min - mid - k >= 0) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        return min + left + k - 1;
    }
}
