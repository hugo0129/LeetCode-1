class Solution {
    public int canCompleteCircuit(int[] gas, int[] cost) {
        if (gas == null || gas.length == 0) return -1;

        int idx = 0;
        // 石油与消耗的差值
        int totalDiff = 0, curDiff = 0;
        for (int i = 0; i < gas.length; i++) {
            totalDiff += gas[i] - cost[i];
            curDiff += gas[i] - cost[i];
            // i到达不了i+1，设置i+1为起点检测
            if (curDiff < 0) {
                idx = i + 1;
                curDiff = 0;
            }
        }

        return totalDiff >= 0 ? idx : -1;
    }
}
