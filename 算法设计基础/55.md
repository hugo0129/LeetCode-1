class Solution {
    public boolean canJump(int[] nums) {
        if (nums == null || nums.length <= 1) return true;

        int target = nums.length - 1, start = target - 1;
        int offset = 1;
        while (start > 0) {
            // start可跳到target，设置target为start，继续向前遍历
            if (nums[start] >= offset) {
                target = start--;
                offset = 1;
            }
            // start跳不到target，向前检测
            else {
                start--;
                offset++;
            }
        }
        return nums[start] >= offset;
    }
}
