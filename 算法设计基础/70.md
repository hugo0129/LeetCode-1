```java
class Solution {
    public int climbStairs(int n) {
        // 当前方式无法正好达到楼顶，返回0
        if (n < 0) return 0;
        // 正好到达楼顶
        if (n == 0) return 1;
        // 递归
        return climbStairs(n - 1) + climbStairs(n - 2);
    }
}
```

```java
class Solution {
    public int climbStairs(int n) {
       if (n <= 0) return 0;

       int[] dp = new int[Math.max(3, n + 1)];
       dp[1] = 1;
       dp[2] = 2;
       for (int i = 3; i <= n; i++) {
           dp[i] = dp[i - 1] + dp[i - 2];
       }
       return dp[n];
    }
}
```
