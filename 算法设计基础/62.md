class Solution {
    public int uniquePaths(int m, int n) {
        if (m <= 0 || n <= 0) return 0;
        int[] dp = new int[n];
        // 第一行初始化，路径只有一条，即起点一直向右
        for (int i = 0; i < n; i++) {
            dp[i] = 1;
        }
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                // 如果是第一列则路径等于上面格子的路径，不变化
                // 否则路径为右边和上边的和
                if (j != 0) dp[j] += dp[j - 1];
            }
        }
        return dp[n - 1];
    }
}
