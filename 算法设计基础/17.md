class Solution {
    public List<String> letterCombinations(String digits) {
        List<String> res = new LinkedList<>();
        if (digits == null || digits.isEmpty()) return res;

        char[][] map = new char[10][];
        map[2] = new char[]{'a', 'b' ,'c'};
        map[3] = new char[]{'d', 'e', 'f'};
        map[4] = new char[]{'g', 'h', 'i'};
        map[5] = new char[]{'j', 'k', 'l'};
        map[6] = new char[]{'m', 'n', 'o'};
        map[7] = new char[]{'p', 'q', 'r', 's'};
        map[8] = new char[]{'t', 'u', 'v'};
        map[9] = new char[]{'w', 'x', 'y', 'z'};

        letterCombinations(digits, 0, "",map, res);
        return res;
    }

    private void letterCombinations(String digits, int start, String s, char[][] map, List<String> res) {
        if (start == digits.length()) {
            res.add(s);
            return;
        }
        for (char c : map[digits.charAt(start) - '0']) {
            letterCombinations(digits, start + 1, s + c, map, res);
        }
    }
}
