class Solution {
    public boolean isMatch(String s, String p) {
        return isMatch(s, p, 0, 0);
    }

    private boolean isMatch(String s, String p, int l, int r) {
        if (l == s.length()) {
            while (r < p.length() && p.charAt(r) == '*') {
                r++;
            }
            return p.length() == r;
        }
        if (p.length() == r) return false;

        if (s.charAt(l) == p.charAt(r) || p.charAt(r) == '?') return isMatch(s, p, l + 1, r + 1);
        if (p.charAt(r) == '*') return isMatch(s, p, l + 1, r) || isMatch(s, p, l + 1, r + 1) || isMatch(s, p, l, r + 1);
        return false;
    }
}
